	\begin{frame}{The rules}
		We cannot:
		\begin{itemize}
			\item Delay scheduled trains
			\item Provide a non-realistic path
		\end{itemize}
		We can:
		\begin{itemize}
			\item Add \textbf{detours}
			\item Slow down the new train
			\item Change the departure time
			\item Add stops, lengthen planned stops
		\end{itemize}
	\end{frame}
